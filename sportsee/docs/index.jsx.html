import styled from "styled-components"

// styled-component
const CardLabel = styled.div`
  background: red;
`

// Component
/**
 * Generate cards
 * @param {*} param0 
 * @returns { HTMLElement } the card
 */
function Card({image, value, type}) {
  return (
      <CardLabel>
          <img></img>
          <div> {value} </div>
          <div> {type} </div>
      </CardLabel>
  )
}

/**
 * Additionne les 2 paramètres 
 * @param {number} a un chiffre
 * @param {number} b un autre chiffre
 * @returns le résultat
 */
function add(a,b) {
  return a+b
}
